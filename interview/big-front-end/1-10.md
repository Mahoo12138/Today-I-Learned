## 1. implement curry()

```js
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length){
      return fn.apply(this, args)
    } else {
      return (...other) => {
        return curried.apply(this, args.concat(other))
      }
    }
  }
}
```
